local bSP=mkBlk(6,"Spectate",1)
do
	local r=mkRow(bSP,1,32);mkL(r,"Enable")
	mkTog(r,function() if specTgt then specOn=true;startSpec() end end,function() specOn=false;stopSpec() end)
	local rs=mkRow(bSP,2,32);mkL(rs,"Target")
	local sL=Instance.new("TextLabel",rs);sL.Size=UDim2.new(0,148,0,20);sL.Position=UDim2.new(0,86,0.5,-10);sL.BackgroundColor3=Color3.fromRGB(20,20,20);sL.BorderSizePixel=1;sL.BorderColor3=Color3.fromRGB(40,40,40);sL.Text="-- ยังไม่ได้เลือก --";sL.TextColor3=Color3.fromRGB(115,115,115);sL.TextSize=11;sL.Font=Enum.Font.Code
	local rl=mkRow(bSP,3,110)
	local pSF=Instance.new("ScrollingFrame",rl);pSF.Size=UDim2.new(1,-10,1,-8);pSF.Position=UDim2.new(0,5,0,4);pSF.BackgroundColor3=Color3.fromRGB(20,20,20);pSF.BorderSizePixel=1;pSF.BorderColor3=Color3.fromRGB(36,36,36);pSF.ScrollBarThickness=4;pSF.ScrollBarImageColor3=Color3.fromRGB(65,65,65);pSF.CanvasSize=UDim2.new(0,0,0,0);pSF.AutomaticCanvasSize=Enum.AutomaticSize.Y
	Instance.new("UIListLayout",pSF).SortOrder=Enum.SortOrder.LayoutOrder
	local function bList()
		for _,c in ipairs(pSF:GetChildren()) do if not c:IsA("UIListLayout") then c:Destroy() end end
		local cnt=0;for _,p in ipairs(Pl:GetPlayers()) do if p~=lp then cnt+=1
			local b=Instance.new("TextButton",pSF);b.Size=UDim2.new(1,0,0,26);b.BackgroundColor3=specTgt==p.Name and Color3.fromRGB(34,50,80) or Color3.fromRGB(24,24,24);b.BorderSizePixel=0;b.LayoutOrder=cnt;b.Text="  "..p.Name;b.TextColor3=specTgt==p.Name and Color3.fromRGB(91,143,255) or Color3.fromRGB(185,185,185);b.TextSize=11;b.Font=Enum.Font.Code;b.TextXAlignment=Enum.TextXAlignment.Left
			b.MouseButton1Click:Connect(function() specTgt=p.Name;sL.Text=p.Name;sL.TextColor3=Color3.fromRGB(255,255,255);bList() end)
		end end
		if cnt==0 then local nl=Instance.new("TextLabel",pSF);nl.Size=UDim2.new(1,0,0,26);nl.BackgroundTransparency=1;nl.Text="  (ไม่มีผู้เล่นอื่น)";nl.TextColor3=Color3.fromRGB(72,72,72);nl.TextSize=11;nl.Font=Enum.Font.Code;nl.TextXAlignment=Enum.TextXAlignment.Left end
	end
	bList()
	task.spawn(function() while true do task.wait(2);bList();if specTgt and not Pl:FindFirstChild(specTgt) then specTgt=nil;specOn=false;stopSpec();sL.Text="-- ยังไม่ได้เลือก --";sL.TextColor3=Color3.fromRGB(115,115,115) end end end)
	Pl.PlayerAdded:Connect(function() task.wait(1);bList() end);Pl.PlayerRemoving:Connect(function(p) if specTgt==p.Name then specTgt=nil;specOn=false;stopSpec();sL.Text="-- ยังไม่ได้เลือก --";sL.TextColor3=Color3.fromRGB(115,115,115) end;task.wait(0.1);bList() end)
	mkNote(bSP,4,"เลือกผู้เล่นก่อนแล้วค่อยเปิด")
end
