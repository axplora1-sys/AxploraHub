-- LOGIC — PLAYER
-----------------------------------------------------------------
local function getHum()
	local c = lp.Character
	return c and c:FindFirstChildOfClass("Humanoid")
end

local function applySpeed()
	local h = getHum(); if h then h.WalkSpeed = speedVal end
end

local function applyJump()
	local h = getHum()
	if h then h.UseJumpPower = true; h.JumpPower = jumpVal end
end

local function startFly()
	local char = lp.Character; if not char then return end
	local root = char:FindFirstChild("HumanoidRootPart")
	local hum  = char:FindFirstChildOfClass("Humanoid")
	if not root or not hum then return end
	hum:ChangeState(Enum.HumanoidStateType.Physics)
	for _, v in pairs(root:GetChildren()) do
		if v:IsA("BodyVelocity") or v:IsA("BodyGyro") then v:Destroy() end
	end
	flyBV = Instance.new("BodyVelocity")
	flyBV.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
	flyBV.Velocity = Vector3.zero; flyBV.Parent = root
	flyBG = Instance.new("BodyGyro")
	flyBG.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
	flyBG.P = 9e4; flyBG.Parent = root
	task.spawn(function()
		while flyOn and char.Parent do
			local md = hum.MoveDirection; local cf = camera.CFrame
			if md.Magnitude > 0 then
				local lm = cf:VectorToObjectSpace(md)
				local dir = (cf.LookVector * -lm.Z) + (cf.RightVector * lm.X)
				if dir.Magnitude > 0 then flyBV.Velocity = dir.Unit * flySpd end
			else flyBV.Velocity = Vector3.zero end
			flyBG.CFrame = cf; task.wait()
		end
	end)
end

local function stopFly()
	local char = lp.Character
	if char then
		local root = char:FindFirstChild("HumanoidRootPart")
		if root then
			for _, v in pairs(root:GetChildren()) do
				if v:IsA("BodyVelocity") or v:IsA("BodyGyro") then v:Destroy() end
			end
		end
		local h = char:FindFirstChildOfClass("Humanoid")
		if h then h:ChangeState(Enum.HumanoidStateType.GettingUp) end
	end
	flyBV = nil; flyBG = nil
end

local function startFling()
	task.spawn(function()
		local ml = 0.1
		while flingOn do
			RunService.Heartbeat:Wait()
			local char = lp.Character
			local hrp  = char and char:FindFirstChild("HumanoidRootPart")
			if hrp then
				local v = hrp.Velocity
				hrp.Velocity = v * 10000 + Vector3.new(0, 10000, 0)
				RunService.RenderStepped:Wait()
				hrp.Velocity = v
				RunService.Stepped:Wait()
				hrp.Velocity = v + Vector3.new(0, ml, 0)
				ml = -ml
			end
		end
	end)
end

-----------------------------------------------------------------
-- LOGIC — TELEPORT
-----------------------------------------------------------------
local function doTeleport()
	if not tpTarget then return end
	local t = Players:FindFirstChild(tpTarget); if not t then return end
	local mc = lp.Character; local tc = t.Character
	if not mc or not tc then return end
	local mr = mc:FindFirstChild("HumanoidRootPart")
	local tr = tc:FindFirstChild("HumanoidRootPart")
	if not mr or not tr then return end
	mr.CFrame = tr.CFrame * CFrame.new(0, 0, 3)
end

task.spawn(function()
	while true do task.wait(0.1); if tpOn then doTeleport() end end
end)

-- LOGIC — PROTECT
-----------------------------------------------------------------
-- Noclip
local noclipConn = nil
local function startNoclip()
	noclipConn = RunService.Stepped:Connect(function()
		local char = lp.Character; if not char then return end
		for _, v in pairs(char:GetDescendants()) do
			if v:IsA("BasePart") then v.CanCollide = false end
		end
	end)
end
local function stopNoclip()
	if noclipConn then noclipConn:Disconnect(); noclipConn = nil end
	local char = lp.Character; if not char then return end
	for _, v in pairs(char:GetDescendants()) do
		if v:IsA("BasePart") then v.CanCollide = true end
	end
end

-- Anti-Fling (ทำให้ velocity ไม่กระโดด)
local antiFlingConn = nil
local function startAntiFling()
	antiFlingConn = RunService.Heartbeat:Connect(function()
		local char = lp.Character
		local hrp  = char and char:FindFirstChild("HumanoidRootPart")
		if hrp and hrp.Velocity.Magnitude > 300 then
			hrp.Velocity = Vector3.zero
		end
	end)
end
local function stopAntiFling()
	if antiFlingConn then antiFlingConn:Disconnect(); antiFlingConn = nil end
end

-- Anti-Aim (หมุน character สุ่มเพื่อรบกวน aim)
local antiAimConn = nil
local function startAntiAim()
	antiAimConn = RunService.RenderStepped:Connect(function()
		local char = lp.Character
		local hrp  = char and char:FindFirstChild("HumanoidRootPart")
		if hrp then
			hrp.CFrame = hrp.CFrame * CFrame.Angles(0, math.random(-180,180) * math.pi/180, 0)
		end
	end)
end
local function stopAntiAim()
	if antiAimConn then antiAimConn:Disconnect(); antiAimConn = nil end
end

	if flyOn      then startFly()   end
	if noclipOn   then startNoclip()   end
	if antiFlingOn then startAntiFling() end
	if antiAimOn  then startAntiAim()  end

local function switchTab(idx)
	activeTab = idx
	-- show/hide blocks
	for ti, blocks in ipairs(tabPages) do
		for _, blk in ipairs(blocks) do
			blk.Visible = (ti == idx)
		end
	end
	-- update tab style
	for i, t in ipairs(tabBtns) do
		t.btn.TextColor3 = (i == idx)
			and Color3.fromRGB(255,255,255)
			or  Color3.fromRGB(90,90,90)
		t.line.Visible = (i == idx)
	end
	-- scroll to top
	Scroll.CanvasPosition = Vector2.new(0,0)
end

for i, t in ipairs(tabBtns) do
	local idx = i
	t.btn.MouseButton1Click:Connect(function() switchTab(idx) end)
end

	local minus = Instance.new("TextButton", r1)
	minus.Size = UDim2.new(0,20,0,20); minus.Position = UDim2.new(0,86,0.5,-10)
	minus.BackgroundColor3 = Color3.fromRGB(40,40,40)
	minus.BorderSizePixel = 1; minus.BorderColor3 = Color3.fromRGB(58,58,58)
	minus.Text = "-"; minus.TextColor3 = Color3.fromRGB(200,200,200)
	minus.TextSize = 14; minus.Font = Enum.Font.Code
	local nb = Instance.new("TextBox", r1)
	nb.Size = UDim2.new(0,44,0,20); nb.Position = UDim2.new(0,110,0.5,-10)
	nb.BackgroundColor3 = Color3.fromRGB(15,15,15)
	nb.BorderSizePixel = 1; nb.BorderColor3 = Color3.fromRGB(40,40,40)
	nb.Text = tostring(def); nb.TextColor3 = Color3.fromRGB(255,255,255)
	nb.TextSize = 12; nb.Font = Enum.Font.Code; nb.ClearTextOnFocus = false
	local plus = Instance.new("TextButton", r1)
	plus.Size = UDim2.new(0,20,0,20); plus.Position = UDim2.new(0,158,0.5,-10)
	plus.BackgroundColor3 = Color3.fromRGB(40,40,40)
	plus.BorderSizePixel = 1; plus.BorderColor3 = Color3.fromRGB(58,58,58)
	plus.Text = "+"; plus.TextColor3 = Color3.fromRGB(200,200,200)
	plus.TextSize = 14; plus.Font = Enum.Font.Code
	-- row slider
	local r2 = makeRow(parent, order+1, 26); makeLbl(r2, "Slider")
	local sbg = Instance.new("Frame", r2)
	sbg.Size = UDim2.new(0,126,0,4); sbg.Position = UDim2.new(0,86,0.5,-2)
	sbg.BackgroundColor3 = Color3.fromRGB(48,48,48); sbg.BorderSizePixel = 0
	local sfill = Instance.new("Frame", sbg)
	sfill.BackgroundColor3 = Color3.fromRGB(91,143,255); sfill.BorderSizePixel = 0
	sfill.Size = UDim2.new((def-mn)/(mx-mn),0,1,0)
	local sk = Instance.new("TextButton", sbg)
	sk.Size = UDim2.new(0,12,0,12); sk.ZIndex = 5
	sk.Position = UDim2.new((def-mn)/(mx-mn),-6,0.5,-6)
	sk.BackgroundColor3 = Color3.fromRGB(91,143,255); sk.BorderSizePixel = 0; sk.Text = ""

	local val = def
	local function set(v)
		val = math.clamp(math.floor(v/stp+0.5)*stp, mn, mx)
		nb.Text = tostring(val)
		local ratio = (val-mn)/(mx-mn)
		sfill.Size = UDim2.new(ratio,0,1,0); sk.Position = UDim2.new(ratio,-6,0.5,-6)
		if onChange then onChange(val) end
	end
	minus.MouseButton1Click:Connect(function() set(val-stp) end)
	plus.MouseButton1Click:Connect(function()  set(val+stp) end)
	nb.FocusLost:Connect(function()
		local n = tonumber(nb.Text); if n then set(n) else nb.Text = tostring(val) end
	end)
	local drag = false
	sk.MouseButton1Down:Connect(function() drag = true end)
	sbg.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 then drag = true end
	end)
	UIS.InputEnded:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 then drag = false end
	end)
	UIS.InputChanged:Connect(function(i)
		if drag and i.UserInputType == Enum.UserInputType.MouseMovement then
			local rx = math.clamp((i.Position.X - sbg.AbsolutePosition.X)/sbg.AbsoluteSize.X,0,1)
			set(mn + rx*(mx-mn))
		end
	end)
end
