local pg = player:WaitForChild("PlayerGui")

local Library = {}

-- กันโหลดซ้ำ
if pg:FindFirstChild("AxploraUI") then
	pg.AxploraUI:Destroy()
end

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "AxploraUI"
gui.ResetOnSpawn = false
gui.IgnoreGuiInset = true
gui.DisplayOrder = 999
gui.Parent = pg

-- เปิด/ปิด
local isVisible = true
gui.Enabled = isVisible

_G.ToggleAxplora = function(state)
	isVisible = state
	gui.Enabled = state
end

-- MAIN
local win = Instance.new("Frame")
win.Size = UDim2.new(0,620,0,460)
win.Position = UDim2.new(0.5,-310,0.5,-230)
win.BackgroundColor3 = Color3.fromRGB(25,25,25)
win.BorderSizePixel = 0
win.Parent = gui

Instance.new("UICorner", win).CornerRadius = UDim.new(0,12)

-- TOPBAR
local top = Instance.new("Frame")
top.Size = UDim2.new(1,0,0,36)
top.BackgroundColor3 = Color3.fromRGB(15,15,15)
top.BorderSizePixel = 0
top.Parent = win

local title = Instance.new("TextLabel")
title.Text = "Axplora Hub_by AH. | v0.0.1 "
title.Size = UDim2.new(1,0,1,0)
title.BackgroundTransparency = 1
title.TextColor3 = Color3.fromRGB(200,50,50)

title.TextSize = 14
title.TextXAlignment = Enum.TextXAlignment.Left
title.Position = UDim2.new(0,12,0,0)
title.Parent = top

-- DRAG
local dragging, dragStart, startPos

local function update(input)
	local delta = input.Position - dragStart
	win.Position = UDim2.new(
		startPos.X.Scale,
		startPos.X.Offset + delta.X,
		startPos.Y.Scale,
		startPos.Y.Offset + delta.Y
	)
end

top.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1
	or input.UserInputType == Enum.UserInputType.Touch then

		dragging = true
		dragStart = input.Position
		startPos = win.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

top.InputChanged:Connect(function(input)
	if dragging then
		update(input)
	end
end)

-- TABS
local tabs = {}

local function selectTab(index)
	for i, t in pairs(tabs) do
		t.page.Visible = (i == index)
	end
end

function Library:AddTab(name)
	local btn = Instance.new("TextButton")
	btn.Text = name
	btn.Size = UDim2.new(1,0,0,32)
	btn.BackgroundTransparency = 1
	btn.TextColor3 = Color3.fromRGB(150,150,150)
	btn.Font = Enum.Font.GothamBold
	btn.TextSize = 12
	btn.Parent = sidebar

	local page = Instance.new("ScrollingFrame")
	page.Size = UDim2.new(1,0,1,0)
	page.BackgroundTransparency = 1
	page.BorderSizePixel = 0
	page.ScrollBarThickness = 4
	page.CanvasSize = UDim2.new(0,0,0,0)
	page.Visible = false
	page.Parent = content

	local layout = Instance.new("UIListLayout", page)
	layout.Padding = UDim.new(0,8)

	layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
		page.CanvasSize = UDim2.new(0,0,0,layout.AbsoluteContentSize.Y + 20)
	end)

	local index = #tabs + 1
	tabs[index] = {btn = btn, page = page}

	btn.MouseButton1Click:Connect(function()
		selectTab(index)
	end)

	if index == 1 then
		selectTab(1)
	end

	local Tab = {}

	function Tab:AddButton(text, callback)
		local b = Instance.new("TextButton")
		b.Text = text
		b.Size = UDim2.new(1,0,0,30)
		b.Parent = page

		b.MouseButton1Click:Connect(function()
			if callback then callback() end
		end)
	end

	function Tab:AddToggle(text, default, callback)
		local state = default or false

		local tog = Instance.new("TextButton")
		tog.Size = UDim2.new(1,0,0,30)
		tog.Text = text .. " : OFF"
		tog.Parent = page

		tog.MouseButton1Click:Connect(function()
			state = not state
			tog.Text = text .. " : " .. (state and "ON" or "OFF")
			if callback then callback(state) end
		end)
	end

	return Tab
end

return Library
