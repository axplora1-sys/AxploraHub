local Players = game:GetService("Players")
local player = Players.LocalPlayer
local pg = player:WaitForChild("PlayerGui")

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "AxploraUI"
gui.ResetOnSpawn = false
gui.Parent = pg

-- MAIN WINDOW
local win = Instance.new("Frame")
win.Size = UDim2.new(0,620,0,460)
win.Position = UDim2.new(0.5,-310,0.5,-230)
win.BackgroundColor3 = Color3.fromRGB(25,25,25)
win.BorderSizePixel = 0
win.Parent = gui

local corner = Instance.new("UICorner")
corner.CornerRadius = UDim.new(0,12)
corner.Parent = win

-- TOPBAR
local top = Instance.new("Frame")
top.Size = UDim2.new(1,0,0,36)
top.BackgroundColor3 = Color3.fromRGB(15,15,15)
top.BorderSizePixel = 0
top.Parent = win

local title = Instance.new("TextLabel")
title.Text = "Axplora Hub"
title.Size = UDim2.new(1,0,1,0)
title.BackgroundTransparency = 1
title.TextColor3 = Color3.fromRGB(192,38,211)
title.Font = Enum.Font.GothamBold
title.TextSize = 14
title.TextXAlignment = Enum.TextXAlignment.Left
title.Position = UDim2.new(0,12,0,0)
title.Parent = top

-- CLOSE
local close = Instance.new("TextButton")
close.Text = "X"
close.Size = UDim2.new(0,30,1,0)
close.Position = UDim2.new(1,-30,0,0)
close.BackgroundTransparency = 1
close.TextColor3 = Color3.fromRGB(200,200,200)
close.Font = Enum.Font.GothamBold
close.TextSize = 14
close.Parent = top

close.MouseButton1Click:Connect(function()
	gui:Destroy()
end)

-- MINIMIZE
local mini = Instance.new("TextButton")
mini.Text = "_"
mini.Size = UDim2.new(0,30,1,0)
mini.Position = UDim2.new(1,-60,0,0)
mini.BackgroundTransparency = 1
mini.TextColor3 = Color3.fromRGB(200,200,200)
mini.Font = Enum.Font.GothamBold
mini.TextSize = 16
mini.Parent = top

local minimized = false
mini.MouseButton1Click:Connect(function()
	minimized = not minimized
	for _,v in pairs(win:GetChildren()) do
		if v ~= top then
			v.Visible = not minimized
		end
	end
end)

-- SIDEBAR (พื้นที่ว่างไว้ใส่ tab ในอนาคต)
local sidebar = Instance.new("Frame")
sidebar.Size = UDim2.new(0,140,1,-36)
sidebar.Position = UDim2.new(0,0,0,36)
sidebar.BackgroundColor3 = Color3.fromRGB(18,18,18)
sidebar.BorderSizePixel = 0
sidebar.Parent = win

local sideCorner = Instance.new("UICorner")
sideCorner.CornerRadius = UDim.new(0,10)
sideCorner.Parent = sidebar

-- CONTENT AREA (พื้นที่หลัก)
local content = Instance.new("Frame")
content.Size = UDim2.new(1,-140,1,-36)
content.Position = UDim2.new(0,140,0,36)
content.BackgroundTransparency = 1
content.Parent = win
